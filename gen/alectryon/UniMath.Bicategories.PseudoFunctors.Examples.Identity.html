C: bicat

f: (C ⟦ a, b ⟧)%cat
g: (C ⟦ b, c ⟧)%cat

((f ◃ id₂ (g · h)%cat) • id₂ (f · (g · h))%cat)
• lassociator f g h =
(lassociator f g h • (id₂ (f · g)%cat ▹ h))
• id₂ (f · g · h)%cat
