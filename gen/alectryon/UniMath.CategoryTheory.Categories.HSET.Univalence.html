(** ** HSET is a univalent_category. *)

Definition hset_id_weq_z_iso (A B : ob HSET) :
  (A = B) ≃ (z_iso A B) :=
  weqcomp (UA_for_HLevels 2 A B) (hset_equiv_weq_z_iso A B).


(** The map [precat_paths_to_iso]
    for which we need to show [isweq] is actually
    equal to the carrier of the weak equivalence we
    constructed above.
    We use this fact to show that that [precat_paths_to_iso]
    is an equivalence.
*)


A, B: SET
idtoiso = pr1 (hset_id_weq_z_iso A B)


A, B: SET
idtoiso = pr1 (hset_id_weq_z_iso A B)

  
A, B: SET
idtoiso ~ pr1 (hset_id_weq_z_iso A B)

  
A, B: SET
p: A = B
idtoiso (idpath A) =
pr1 (hset_id_weq_z_iso A A) (idpath A)

  
A, B: SET
p: A = B
identity A = (λ x : pr1 A, x)

  
A, B: SET
p: A = B
identity A = (λ x : pr1 A, x)
 
pr1: UU
pr2: isaset pr1
B: SET
p: pr1,, pr2 = B
x: pr1,, pr2
identity (pr1,, pr2) x = x

    apply idpath.
Defined.


A, B: SET
isweq idtoiso


A, B: SET
isweq idtoiso

  
A, B: SET
isweq (pr1 (hset_id_weq_z_iso A B))

  apply (pr2 (hset_id_weq_z_iso A B)).
Defined.

Definition category_HSET : category := make_category HSET has_homsets_HSET.


is_univalent category_HSET


is_univalent category_HSET

  
a, b: category_HSET
isweq (λ p : a = b, idtoiso p)

  apply (is_weq_precat_paths_to_iso_hset a b).
Defined.

Definition HSET_univalent_category : univalent_category := _ ,, is_univalent_HSET.
