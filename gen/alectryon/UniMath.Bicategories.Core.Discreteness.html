Definition isaprop_2cells
           (B : bicat)
  : UU
  := ∏ (x y : B)
       (f g : x --> y)
       (α β : f ==> g),
     α = β.

Definition is_discrete_bicat
           (B : bicat)
  : UU
  := is_univalent_2_1 B
     ×
     locally_groupoid B
     ×
     isaprop_2cells B.

Definition is_locally_posetal_bicat
           (B : bicat)
  : UU
  := is_univalent_2_1 B
     ×
     isaprop_2cells B.
