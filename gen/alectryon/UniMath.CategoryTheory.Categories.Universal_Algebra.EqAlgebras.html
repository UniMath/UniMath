Local Open Scope sorted_scope.

disp_cat (category_algebras σ)

disp_cat (category_algebras σ)

  
σ: eqspec
category_algebras σ → UU

  exact (λ A, is_eqalgebra (hSetalgebra_to_algebra A)).
Defined.


σ: eqspec
is_univalent_disp eqalg_disp


σ: eqspec
is_univalent_disp eqalg_disp

  
σ: eqspec
∏ x : category_algebras σ,
isaprop
  ((λ A : category_algebras σ, is_eqalgebra A) x)

  
σ: eqspec
A: category_algebras σ
isT, isT': is_eqalgebra A
isofhlevel 0 (isT = isT')

  
σ: eqspec
A: category_algebras σ
isT, isT': is_eqalgebra A
∏ t : equations σ,
isaprop ((λ t0 : equations σ, holds A (geteq t0)) t)

  
σ: eqspec
A: category_algebras σ
isT, isT': is_eqalgebra A
eq: equations σ
isaprop ((λ t : equations σ, holds A (geteq t)) eq)

  
σ: eqspec
A: category_algebras σ
isT, isT': is_eqalgebra A
eq: equations σ
∏ t : assignment (pr1 A) (variables σ),
isaprop
  ((λ t0 : assignment (pr1 A) (variables σ),
    fromterm (ops A) t0 (eqsort (geteq eq))
      (lhs (geteq eq)) =
    fromterm (ops A) t0 (eqsort (geteq eq))
      (rhs (geteq eq))) t)

  
σ: eqspec
A: category_algebras σ
isT, isT': is_eqalgebra A
eq: equations σ
α: assignment (pr1 A) (variables σ)
p, p': fromterm (ops A) α (eqsort (geteq eq))
  (lhs (geteq eq)) =
fromterm (ops A) α (eqsort (geteq eq))
  (rhs (geteq eq))
isofhlevel 0 (p = p')

  apply (pr1 A).
Qed.

Definition category_eqalgebras : category := total_category eqalg_disp.


σ: eqspec
is_univalent category_eqalgebras


σ: eqspec
is_univalent category_eqalgebras

  exact (@is_univalent_total_category
           (category_algebras σ) eqalg_disp (is_univalent_category_algebras σ) is_univalent_eqalg_disp).
Qed.
