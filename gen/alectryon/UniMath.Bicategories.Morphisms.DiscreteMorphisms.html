(**
 1. Conservative 1-cells
 *)
Definition conservative_1cell
           {B : bicat}
           {a b : B}
           (f : a --> b)
  : UU
  := ∏ (x : B)
       (g₁ g₂ : x --> a)
       (α : g₁ ==> g₂),
     is_invertible_2cell (α ▹ f)
     →
     is_invertible_2cell α.

Definition conservative_1cell_reflect_iso
           {B : bicat}
           {a b : B}
           {f : a --> b}
           (Hf : conservative_1cell f)
           {x : B}
           {g₁ g₂ : x --> a}
           (α : g₁ ==> g₂)
           (Hα : is_invertible_2cell (α ▹ f))
  : is_invertible_2cell α
  := Hf x g₁ g₂ α Hα.


B: bicat
a, b: B
f: B ⟦ a, b ⟧
isaprop (conservative_1cell f)


B: bicat
a, b: B
f: B ⟦ a, b ⟧
isaprop (conservative_1cell f)

  
B: bicat
a, b: B
f: B ⟦ a, b ⟧
t: B
t0, t1: B ⟦ t, a ⟧
t2: t0 ==> t1
t3: is_invertible_2cell (t2 ▹ f)
isofhlevel 1
  ((λ _ : is_invertible_2cell (t2 ▹ f),
    is_invertible_2cell t2) t3)

  apply isaprop_is_invertible_2cell.
Qed.

(**
 4. Discrete 1-cells
 *)
Definition discrete_1cell
           {B : bicat}
           {a b : B}
           (f : a --> b)
  : UU
  := faithful_1cell f × conservative_1cell f.

 apply isaprop_conservative_1cell.
Qed.
