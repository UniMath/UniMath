(* useful lemma in a lot of proofs where we transport
   arrows / three morphisms *)

A, B: UU
P: A → B → UU
a, a': A
e: a = a'
xs: ∑ b : B, P a b
pr1 (transportf (λ x : A, ∑ b : B, P x b) e xs) =
pr1 xs

A, B: UU
P: A → B → UU
a, a': A
e: a = a'
xs: ∑ b : B, P a b
transportf (λ _ : A, B) e (pr1 xs) = pr1 xs

  
A, B: UU
P: A → B → UU
a, a': A
e: a = a'
xs: ∑ b : B, P a b
idfun B (pr1 xs) = pr1 xs

  reflexivity.
Qed.
