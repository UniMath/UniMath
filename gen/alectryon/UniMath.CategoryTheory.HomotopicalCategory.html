Definition two_of_six {C : category} (W : morph_class C) :=
  ∏ (x y z t : C)
    (f  : x --> y)
    (g  : y --> z)
    (h  : z --> t)
    (gf : W x z (f · g))
    (hg  : W y t (g · h)),
    (W x y f)
      × (W y z g)
      × (W z t h)
      × (W x t (f · g · h)).
