(** *** Transport a morphism using funextfun *)


X: UU
C: precategory_ob_mor
F, F': X → C
H: ∏ x : X, F x = F' x
x: X
c: C
f: C ⟦ F x, c ⟧
transportf (λ x0 : X → C, C ⟦ x0 x, c ⟧)
  (funextfun F F' H) f =
transportf (λ x0 : C, C ⟦ x0, c ⟧) (H x) f


X: UU

  (funextfun F F' H) f =
transportf (λ x0 : C, C ⟦ x0, c ⟧) (H x) f

  exact (@transportf_funextfun X (ob C) (λ x0 : C, x0 --> c) F F' H x f).
Qed.
