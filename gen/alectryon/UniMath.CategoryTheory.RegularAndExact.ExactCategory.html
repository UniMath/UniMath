(** * 1. Internal relations *)
Definition internal_relation
           {C : category}
           (x : C)
  : UU
  := ∑ (r : C)
       (s t : r --> x),
     ∏ (w : C)
       (f g : w --> r),
     f · s = g · s
     →
     f · t = g · t
     →
     f = g.

Definition make_internal_relation
           {C : category}
           (x : C)
           (r : C)
           (s t : r --> x)
           (p : ∏ (w : C)

                f · s = g · s
                →
                f · t = g · t
                →
                f = g)
  : internal_relation x
  := r ,, s ,, t ,, p.
