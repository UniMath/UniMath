Definition SigmaMonoid_characteristic_equation {V : category} {Mon_V : monoidal V} {H : V ⟶ V}
    (x : V) (η : V ⟦ monoidal_unit Mon_V, x ⟧)
    (μ : V ⟦ x ⊗_{ Mon_V} x, x ⟧) (τ :  V ⟦ H x, x ⟧)
    (st : V ⟦ x ⊗_{ Mon_V} H x, H (x ⊗_{ Mon_V} x) ⟧) : UU
    := st · #H μ · τ = x ⊗^{Mon_V}_{l} τ · μ.
