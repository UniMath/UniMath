  Definition kleisli_cocone
    : UU
    := ∑ (x : B)
         (f : z --> x)
         (γ : h · f ==> f),
       (η ▹ f • γ = lunitor _)
       ×
       (rassociator _ _ _ • (_ ◃ γ) • γ = (μ ▹ f) • γ).

  Definition make_kleisli_cocone
             {x : B}
             (f : z --> x)
             (γ : h · f ==> f)
             (fη : η ▹ f • γ = lunitor _)
             (fμ : rassociator _ _ _ • (_ ◃ γ) • γ = (μ ▹ f) • γ)
    : kleisli_cocone
    := x ,, f ,, γ ,, fη ,, fμ.

  Coercion kleisli_cocone_ob (q : kleisli_cocone) : B := pr1 q.

             (q₁ q₂ : kleisli_cocone)
