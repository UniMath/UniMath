(**
 1. Biinitial object
 *)
Section OpCellBiinitial.
  Context {B : bicat}
          (b : B)
          (H : is_biinitial b).

  
B: bicat
b: B
H: is_biinitial b
is_biinitial b

  
B: bicat
b: B
H: is_biinitial b
is_biinitial b

    
B: bicat
b: B
H: is_biinitial b
biinitial_1cell_property b
B: bicat
b: B
H: is_biinitial b
∏ Y : op2_bicat B, biinitial_2cell_property b Y
B: bicat
b: B
H: is_biinitial b
∏ Y : op2_bicat B, biinitial_eq_property b Y

    
B: bicat
b: B
H: is_biinitial b
biinitial_1cell_property b
 exact (λ y, is_biinitial_1cell_property H y).
    
B: bicat
b: B
H: is_biinitial b
∏ Y : op2_bicat B, biinitial_2cell_property b Y
 exact (λ y f g, is_biinitial_2cell_property H _ g f).
    
B: bicat
b: B
H: is_biinitial b
∏ Y : op2_bicat B, biinitial_eq_property b Y
 exact (λ y f g α β, is_biinitial_eq_property H _ _ _ α β).
  Defined.
End OpCellBiinitial.

Definition op_cell_biinitial
           {B : bicat}
           (I : biinitial_obj B)
  : biinitial_obj (op2_bicat B)
  := pr1 I ,, op_cell_is_biinitial _ (pr2 I).
