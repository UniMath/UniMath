Definition identityPathOver {X:Type} {x:X} {Y : X -> Type} (y : Y x)

Definition pathOverIdpath {X:Type} {x:X} {Y : X -> Type} (y y' : Y x)
  : PathOver (idpath x) y y' = (y = y')

  : y = y' -> PathOver (idpath x) y y'
