(** ** Forgetful functor. *)

Definition cgraph_forget_map (G : cgraph) : hSet
  := make_hSet (node G) (isaset_node G).

Definition cgraph_forget_data
  : functor_data cgraph_precategory_ob_mor hset_precategory_ob_mor
  := @make_functor_data
       cgraph_category HSET
       cgraph_forget_map
       (λ G H : cgraph, onnode).

Qed.
