Require UniMath.Algebra.IteratedBinaryOperations.
Require UniMath.Foundations.NaturalNumbers.
Require UniMath.Algebra.IteratedBinaryOperations.
Require UniMath.Combinatorics.FiniteSets.

Module Test_assoc.

  Import UniMath.Algebra.IteratedBinaryOperations.
  Import UniMath.Foundations.NaturalNumbers.

  (* verify that our associativity matches that of the parser, without an extra "1" *)

  Local Notation "[]" := Lists.nil (at level 0, format "[]").
  Local Infix "::" := cons.

  Section Test.
    Context (X:UU) (e:X) (op:binop X) (w x y z:X).
    
X: UU
e: X
op: binop X
w, x, y, z: X
iterop_list e op [] = e
 reflexivity. Qed.
    
X: UU
e: X
op: binop X
w, x, y, z: X
iterop_list e op (x :: []) = x
 reflexivity. Qed.
    
X: UU
e: X
op: binop X
w, x, y, z: X
iterop_list e op (x :: y :: []) = op x y
 reflexivity. Qed.
    
X: UU
e: X
op: binop X
w, x, y, z: X
iterop_list e op (w :: x :: y :: z :: []) =
op w (op x (op y z))
 reflexivity. Qed.
  End Test.

  Local Open Scope stn.

  Open Scope multmonoid.

  
∏ (M : monoid) (f : ⟦ 3 ⟧ → M),
iterop_seq_mon (3,, f) =
(f (● 0%nat) + f (● 1%nat) + f (● 2))%addmonoid

  
∏ (M : monoid) (f : ⟦ 3 ⟧ → M),
iterop_seq_mon (3,, f) =
(f (● 0%nat) + f (● 1%nat) + f (● 2))%addmonoid
 reflexivity. Defined.

  
∏ (M : monoid) (f : ⟦ 3 ⟧ → Sequence M),
iterop_seq_seq_mon (3,, f) =
(iterop_seq_mon (f (● 0%nat)) +
 iterop_seq_mon (f (● 1%nat)) +
 iterop_seq_mon (f (● 2)))%addmonoid

  
∏ (M : monoid) (f : ⟦ 3 ⟧ → Sequence M),
iterop_seq_seq_mon (3,, f) =
(iterop_seq_mon (f (● 0%nat)) +
 iterop_seq_mon (f (● 1%nat)) +
 iterop_seq_mon (f (● 2)))%addmonoid
 reflexivity. Defined.

  (* demonstrate that the Coq parser is left-associative with "*" *)
  
∏ (M : monoid) (x y z : M),
(x + y + z)%addmonoid = (x + y + z)%addmonoid
 
∏ (M : monoid) (x y z : M),
(x + y + z)%addmonoid = (x + y + z)%addmonoid
 reflexivity. Defined.
  
∏ (M : monoid) (x y z : M),
(x + y + z)%addmonoid = (x + (y + z))%addmonoid
 
∏ (M : monoid) (x y z : M),
(x + y + z)%addmonoid = (x + (y + z))%addmonoid
 apply assocax. Defined.

  (* demonstrate that the Coq parser is left-associative with "+" *)
  Local Open Scope addmonoid.
  
∏ (M : monoid) (x y z : M), x + y + z = x + y + z
 
∏ (M : monoid) (x y z : M), x + y + z = x + y + z
 reflexivity. Defined.
  
∏ (M : monoid) (x y z : M), x + y + z = x + (y + z)
 
∏ (M : monoid) (x y z : M), x + y + z = x + (y + z)
 apply assocax. Defined.

End Test_assoc.
