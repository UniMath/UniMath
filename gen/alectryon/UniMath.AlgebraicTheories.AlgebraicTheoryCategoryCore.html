    {T T' : indexed_set_cat nat}

    (var' : ∏ n i, var_ax T' n i)

    (subst' : ∏ m n f g, subst_ax T' m n f g)
    (m n : nat)
    (f : T m)
    (g : stn m → T n)
    : UU
    := F n (subst m n f g) = subst' m n (F m f) (λ i, F n (g i)).
