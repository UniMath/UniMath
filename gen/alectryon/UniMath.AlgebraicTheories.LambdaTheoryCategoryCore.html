  Definition app_ax
    (T : algebraic_theory)
    (n : nat)
    (f : T n)
    : UU
    := T (S n).

  Definition abs_ax
    (T : algebraic_theory)
    (n : nat)
    (f : T (S n))
    : UU
    := T n.

  Let mor_app_ax'
    {T T' : algebraic_theory}
    (F : algebraic_theory_morphism T T')
    (app : ∏ n f, app_ax T n f)
    (app' : ∏ n f, app_ax T' n f)
    (n : nat)
    (f : T n)
    : UU
    := F (S n) (app n f) = app' n (F n f).

  Let mor_abs_ax'
    {T T' : algebraic_theory}
    (F : algebraic_theory_morphism T T')
    (abs : ∏ n f, abs_ax T n f)
    (abs' : ∏ n f, abs_ax T' n f)
    (n : nat)
    (f : T (S n))
    : UU
    := F n (abs n f) = abs' n (F (S n) f).

    {T : algebraic_theory_data}
    {m n : nat}
    (f : T (S m))
    (g : stn m → T n)

    (L : lambda_theory_data_cat)
    (m n : nat)

    (L : lambda_theory_data_cat)
    (m n : nat)
    (f : data_theory L (S m))
    (g : stn m → data_theory L n)
