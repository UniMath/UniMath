Definition embedding {σ : signature} (B A : algebra σ) :=
  ∑ (i : B s→ A), (∏ s, isincl (i s)) × (ishom i).

Definition make_embedding {σ : signature} {B A : algebra σ}
  (i : B s→ A)
  (ishom : ishom i)
  (isincl : ∏ s, isincl (i s))
  : embedding B A
  := (i,,(isincl,,ishom)).

Definition make_embedding' {σ : signature} {B A : algebra σ}
  (i : B ↷ A)
  (isincl : ∏ s, isincl (i s))
  : embedding B A
  := (hom2fun i,,(isincl,,hom2axiom i)).
