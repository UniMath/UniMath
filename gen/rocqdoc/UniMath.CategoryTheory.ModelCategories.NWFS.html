Definition functorial_factorization (C : category) := section_disp (three_disp C).
Definition fact_section {C : category} (F : functorial_factorization C)
    := section_disp_data_from_section_disp F.
Definition fact_functor {C : category} (F : functorial_factorization C) : arrow C ‚ü∂ three C :=
    section_functor F.
Coercion fact_functor : functorial_factorization >-> functor.

Lemma functorial_factorization_splits_face_map_1 {C : category} (F : functorial_factorization C) :
    F ‚àô face_map_1 = functor_identity (arrow C).
Proof.
  apply functor_eq; trivial.
  apply homset_property.
Defined.

Definition fact_L {C : category} (F : functorial_factorization C) : arrow C ‚ü∂ arrow C :=
    F ‚àô face_map_2.
Definition fact_R {C : category} (F : functorial_factorization C) : arrow C ‚ü∂ arrow C :=
    F ‚àô face_map_0.

Lemma LR_compatibility {C : category} (F : functorial_factorization C) :
    ‚àè (f : arrow C), arrow_mor (fact_L F f) ¬∑ arrow_mor (fact_R F f) = arrow_mor f.
Proof.
  intro.
  exact (three_comp _).
Defined.

Definition Œ¶ {C : category} (F : functorial_factorization C) :
    (fact_L F) ‚üπ (functor_identity (arrow C)) :=
  pre_whisker F (c_21 C).

Definition Œõ {C : category} (F : functorial_factorization C) :
    (functor_identity (arrow C)) ‚üπ (fact_R F) :=
  pre_whisker F (c_10 C).

Definition R_monad_data {C : category} (F : functorial_factorization C)
  (Œ† : (fact_R F) ‚àô (fact_R F) ‚üπ (fact_R F))
  : disp_Monad_data (fact_R F)
  := Œ† ,, Œõ F.

Definition R_monad {C : category} (F : functorial_factorization C)
    (Œ† : (fact_R F) ‚àô (fact_R F) ‚üπ (fact_R F))
    (R : disp_Monad_laws (R_monad_data F Œ†)) : Monad (arrow C) :=
  (fact_R F ,, R_monad_data F Œ†,, R).

Definition L_monad_data {C : category} (F : functorial_factorization C)
  (Œ£ : (fact_L F) ‚üπ (fact_L F) ‚àô (fact_L F))
  : @disp_Monad_data (op_cat (arrow C)) (functor_opp (fact_L F))
  := op_nt Œ£ ,, op_nt (Œ¶ F).

Definition L_monad {C : category} (F : functorial_factorization C)
    (Œ£ : (fact_L F) ‚üπ (fact_L F) ‚àô (fact_L F))
    (L : disp_Monad_laws (L_monad_data F Œ£)) : Monad (op_cat (arrow C)) :=
  (functor_opp (fact_L F) ,, L_monad_data F Œ£,, L).

Definition nwfs (C : category) :=
    ‚àë (F : functorial_factorization C) (Œ£ : (fact_L F) ‚üπ (fact_L F) ‚àô (fact_L F)) (Œ† : (fact_R F) ‚àô (fact_R F) ‚üπ (fact_R F)) ,
    (disp_Monad_laws (L_monad_data F Œ£)) √ó (disp_Monad_laws (R_monad_data F Œ†)).

Definition make_nwfs {C : category} (F : functorial_factorization C)
    (Œ£ : (fact_L F) ‚üπ (fact_L F) ‚àô (fact_L F)) (L : disp_Monad_laws (L_monad_data F Œ£))
    (Œ† : (fact_R F) ‚àô (fact_R F) ‚üπ (fact_R F)) (R : disp_Monad_laws (R_monad_data F Œ†))
        : nwfs C.
Proof.
  exists F, Œ£, Œ†.
  exact (make_dirprod L R).
Defined.

Definition nwfs_Œ£ {C : category} (n : nwfs C) := pr12 n.

Definition nwfs_R_monad {C : category} (n : nwfs C) := R_monad (nwfs_fact n) (nwfs_Œ† n) (nwfs_Œ†_laws n).
Definition nwfs_L_monad {C : category} (n : nwfs C) := L_monad (nwfs_fact n) (nwfs_Œ£ n) (nwfs_Œ£_laws n).

Definition fact_mor {C : category} (F F' : functorial_factorization C) :=

Lemma fact_mor_whisker_d1_is_id {C : category} {F F' : functorial_factorization C}

    post_whisker Œ± face_map_1 = nat_trans_id (functor_identity _).

Definition nwfs_L_mor {C : category} {n n' : nwfs C}
    (Œ± : fact_mor n n') : (nwfs_L_monad n') ‚üπ (nwfs_L_monad n) :=

Definition nwfs_R_mor {C : category} {n n' : nwfs C}
    (Œ± : fact_mor n n') : (nwfs_R_monad n) ‚üπ (nwfs_R_monad n') :=

Definition nwfs_mor_axioms {C : category} (n n' : nwfs C) (Œ± : fact_mor n n') :=
    Monad_Mor_laws (nwfs_L_mor Œ±) √ó Monad_Mor_laws (nwfs_R_mor Œ±).

Lemma isaprop_nwfs_mor_axioms {C : category} (n n' : nwfs C) (Œ± : fact_mor n n') :
  isaprop (nwfs_mor_axioms n n' Œ±).

Qed.

Definition nwfs_mor {C : category} (n n' : nwfs C) :=

Definition nwfs_precategory_ob_mor : precategory_ob_mor.
Proof.
  use make_precategory_ob_mor.
  - exact (nwfs C).
  - intros n n'.
    exact (nwfs_mor n n').
Defined.

Definition fact_id_is_nwfs_mor (n : nwfs C) : nwfs_mor_axioms n n (section_nat_trans_id (nwfs_fact n)).
Proof.
  split.
  - assert (H : nwfs_L_mor (section_nat_trans_id (nwfs_fact n)) = nat_trans_id (nwfs_L_monad n)).
    {
      use nat_trans_eq; [apply homset_property|].
      intro.
      apply subtypePath; [intro; apply homset_property|].
      apply pathsdirprod; cbn; trivial.
    }
    rewrite H.
    apply identity_Monad_Mor_laws.
  - assert (H : nwfs_R_mor (section_nat_trans_id (nwfs_fact n)) = nat_trans_id (nwfs_R_monad n)).
    {
      use nat_trans_eq; [apply homset_property|].
      intro.
      apply subtypePath; [intro; apply homset_property|].
      apply pathsdirprod; cbn; trivial.
    }
    rewrite H.
    apply identity_Monad_Mor_laws.
Qed.

Definition nwfs_mor_id (n : nwfs C) : nwfs_mor n n.
Proof.
  use tpair.
  - exact (section_nat_trans_id (nwfs_fact n)).
  - exact (fact_id_is_nwfs_mor n).
Defined.

Definition nwfs_mor_comp (n n' n'' : nwfs C) :
    (nwfs_mor n n') -> (nwfs_mor n' n'') -> (nwfs_mor n n'').
Proof.
  intros Œ± Œ±'.

  use tpair.
  - exact (section_nat_trans_comp (fact_mor_from_nwfs_mor Œ±) (fact_mor_from_nwfs_mor Œ±')).
  - split.
    * assert (nwfs_L_mor (section_nat_trans_comp (pr1 Œ±) (pr1 Œ±')) =
              nat_trans_comp _ _ _ (nwfs_L_monad_mor Œ±') (nwfs_L_monad_mor Œ±)) as H.
      {
        simpl.
        use nat_trans_eq.
        -
          exact (homset_property (op_cat (arrow C))).
        - intro x; simpl in x.
          apply subtypePath; [intro; apply homset_property|].
          simpl.
          apply pathsdirprod; cbn.
          * now rewrite id_left.
          * unfold three_mor11.
            simpl.
            unfold mor_disp; simpl.
            rewrite pr1_transportf.
            rewrite transportf_const.
            trivial.
      }
      unfold fact_mor_from_nwfs_mor.
      rewrite H.
      exact (composition_Monad_Mor_laws (nwfs_L_monad_mor Œ±') (nwfs_L_monad_mor Œ±)).
    * assert (nwfs_R_mor (section_nat_trans_comp (pr1 Œ±) (pr1 Œ±')) =
              nat_trans_comp _ _ _ (nwfs_R_monad_mor Œ±) (nwfs_R_monad_mor Œ±')) as H.
      {
        simpl.
        use nat_trans_eq.
        -
          exact (homset_property (arrow C)).
        - intro x; simpl in x.
          apply subtypePath; [intro; apply homset_property|].
          simpl.
          apply pathsdirprod; cbn.
          * unfold three_mor11.
            simpl.
            unfold mor_disp; simpl.
            rewrite pr1_transportf.
            rewrite transportf_const.
            trivial.
          * now rewrite id_left.
      }
      unfold fact_mor_from_nwfs_mor.
      rewrite H.
      exact (composition_Monad_Mor_laws (nwfs_R_monad_mor Œ±) (nwfs_R_monad_mor Œ±')).
Defined.

Definition nwfs_precategory_data : precategory_data.
Proof.
  use make_precategory_data.
  - exact nwfs_precategory_ob_mor.
  - exact nwfs_mor_id.
  - exact nwfs_mor_comp.
Defined.

Definition nwfs_is_precategory : is_precategory nwfs_precategory_data.
Proof.
  use make_is_precategory_one_assoc; intros.
  - apply subtypePath; [intro; apply isaprop_nwfs_mor_axioms|cbn].
    apply section_nat_trans_id_left.
  - apply subtypePath; [intro; apply isaprop_nwfs_mor_axioms|cbn].
    apply section_nat_trans_id_right.
  - apply subtypePath; [intro; apply isaprop_nwfs_mor_axioms|cbn].
    apply section_nat_trans_assoc.
Qed.

Definition NWFS_precat : precategory :=
    (nwfs_precategory_data,, nwfs_is_precategory).

Lemma has_homsets_NWFS : has_homsets NWFS_precat.
Proof.
  intros n n' .
  use isaset_nwfs_mor.
Qed.

Definition NWFS : category := (NWFS_precat,, has_homsets_NWFS).
