  Context {B : bicat}
          {x : B}
          (dm₁ dm₂ : disp_mnd B x)
          (α : distr_law dm₁ dm₂).

  Let m₁ : mnd B := x ,, dm₁.
  Let m₂ : mnd B := x ,, dm₂.

  Let f : x --> x := endo_of_mnd m₁.
  Let g : x --> x := endo_of_mnd m₂.

  Let η₁ : id₁ _ ==> f := unit_of_mnd m₁.
  Let η₂ : id₁ _ ==> g := unit_of_mnd m₂.

  Let μ₁ : f · f ==> f := mult_of_mnd m₁.
  Let μ₂ : g · g ==> g := mult_of_mnd m₂.

  Let ηc : id₁ x ==> f · g
    := linvunitor _ • (η₁ ▹ _) • (_ ◃ η₂).

  Let μc
    : f · g · (f · g) ==> f · g
    := rassociator _ _ _
       • (f ◃ lassociator _ _ _)
       • (f ◃ (α ▹ g))
       • (f ◃ rassociator _ _ _)
       • lassociator _ _ _
       • (μ₁ ▹ _)
       • (_ ◃ μ₂).
