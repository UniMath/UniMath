  Defined.

  Lemma empty_is_presheaf
    : is_presheaf empty_presheaf_data.
  Proof.
    use make_is_presheaf;
      intros;
      apply fromempty;
      assumption.
  Qed.

  Definition empty_presheaf
    : presheaf L
    := make_presheaf
      empty_presheaf_data
      empty_is_presheaf.

End EmptyPresheaf.
