1. Opposite Of Weak Equivalence

Lemma opp_is_weak_equiv
  {C D : category} {F : functor C D} (F_weq : is_weak_equiv F)
  : is_weak_equiv (functor_op F).
Proof.
  split.
  - apply opp_functor_essentially_surjective.
    exact (pr1 F_weq).
  - apply opp_functor_fully_faithful.
    exact (pr2 F_weq).
Qed.
