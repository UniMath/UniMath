1. Structure of hyperdoctrines

Definition preorder_hyperdoctrine
  : UU
  := ‚àë (C : category)
       (D : disp_cat C),
     Terminal C
     √ó
     BinProducts C
     √ó
     cleaving D
     √ó
     locally_propositional D.

Definition make_preorder_hyperdoctrine
           (C : category)
           (D : disp_cat C)
           (T : Terminal C)
           (BP : BinProducts C)
           (HD : cleaving D)
           (HD' : locally_propositional D)
  : preorder_hyperdoctrine
  := C ,, D ,, T ,, BP ,, HD ,, HD'.

Definition hyperdoctrine
  : UU
  := ‚àë (C : category)
       (D : disp_cat C),
     Terminal C
     √ó
     BinProducts C
     √ó
     cleaving D
     √ó
     locally_propositional D
     √ó
     is_univalent_disp D.

Definition make_hyperdoctrine
           (C : category)
           (D : disp_cat C)
           (T : Terminal C)
           (BP : BinProducts C)
           (HD : cleaving D)

           (HD'' : is_univalent_disp D)
