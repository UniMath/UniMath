3. Pairing partial setoid morphisms

    repeat split.
    - unfold partial_setoid_mor_dom_defined_law.
      cbn ; hypersimplify.
      pose (w := œÄ‚ÇÇ (œÄ‚ÇÅ (tm_var ((ùüô √óh W) √óh X √óh Y)))).
      pose (x := œÄ‚ÇÅ (œÄ‚ÇÇ (tm_var ((ùüô √óh W) √óh X √óh Y)))).
      pose (y := œÄ‚ÇÇ (œÄ‚ÇÇ (tm_var ((ùüô √óh W) √óh X √óh Y)))).
      fold w x y.
      do 2 use forall_intro.
      use impl_intro.
      do 2 use weaken_right.
      use (partial_setoid_mor_dom_defined œà w y).

      + use weaken_left.
        use (partial_setoid_mor_cod_defined œÜ w x).
        apply hyperdoctrine_hyp.
      + use weaken_right.
        use (partial_setoid_mor_cod_defined œà w y).
        apply hyperdoctrine_hyp.
    - unfold partial_setoid_mor_eq_defined_law.
      pose (T := W).
      pose (T' := prod_partial_setoid X Y).
      pose (w‚ÇÅ := œÄ‚ÇÇ (œÄ‚ÇÅ (œÄ‚ÇÅ (œÄ‚ÇÅ (tm_var ((((ùüô √óh T) √óh T) √óh T') √óh T')))))).

      do 4 use forall_intro.
      use impl_intro.
      use weaken_right.

          apply hyperdoctrine_hyp.
        * use weaken_left.
          use weaken_right.
          use eq_in_prod_partial_setoid_l.
          apply hyperdoctrine_hyp.
        * use weaken_right.
          apply hyperdoctrine_hyp.
      + use hyp_sym.
        use hyp_ltrans.
        use weaken_right.
        use (partial_setoid_mor_eq_defined œà).
        * exact w‚ÇÅ.
        * exact (œÄ‚ÇÇ xy‚ÇÅ).
        * use weaken_right.
          use weaken_left.
          apply hyperdoctrine_hyp.

          use eq_in_prod_partial_setoid_r.
          apply hyperdoctrine_hyp.
        * use weaken_left.
          apply hyperdoctrine_hyp.
    - unfold partial_setoid_mor_unique_im_law.

      pose (z‚ÇÇ := œÄ‚ÇÇ (tm_var (((ùüô √óh T) √óh T') √óh T'))).
      unfold T, T' in *.
      fold x z‚ÇÅ z‚ÇÇ.
      hypersimplify.
      do 3 use forall_intro.
      use impl_intro.
      use weaken_right.
      use impl_intro.
      use eq_in_prod_partial_setoid.
      + use hyp_rtrans.
        use weaken_left.
        use hyp_sym.
        use hyp_rtrans.
        use weaken_left.
        use (partial_setoid_mor_unique_im œÜ).

        use weaken_right.

        use hyp_ltrans.
        use weaken_right.

        * exact x.
        * use weaken_right.
          apply hyperdoctrine_hyp.
        * use weaken_left.
          apply hyperdoctrine_hyp.
    - unfold partial_setoid_mor_hom_exists_law.
      pose (T := W).
      pose (T' := prod_partial_setoid X Y).
      pose (x := œÄ‚ÇÇ (tm_var (ùüô √óh T))).

      unfold T, T' in *.

      hypersimplify.

             (partial_setoid_mor_hom_exists œÜ (hyperdoctrine_hyp _))).

      use (exists_elim
